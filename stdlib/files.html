SysFiles (I/O)

The sysfiles module (also aliased as fs) provides comprehensive tools for file system interaction, including reading, writing, and directory management.

Importing

include sysfiles
# Or use the short alias
include sysfiles as fs

File Operations

Reading and Writing

# Read entire file as string
let content = fs.read("data.txt")
say(content)

# Write string to file (overwrites existing)
fs.write("output.txt", "Hello Mintas!")

# Append string to end of file
fs.append("log.txt", "New log entry\n")

Management

# Check if file or directory exists
if (fs.exists("config.json")):
    say("Config found")
end

# Copy file
fs.copy("source.txt", "backup.txt")

# Move or Rename file
fs.move("old_name.txt", "new_name.txt")

# Delete file or directory
fs.delete("temp_file.txt")

# Create an empty file
fs.touch("new_file.txt")

Directory Operations

# Create directory (and parents if needed)
fs.mkdir("path/to/project")

# List files in directory
let files = fs.list("./docs")
for (file in files):
    say("Found: " + file)
end

# Remove directory
fs.rmdir("old_folder")

# Copy entire directory recursively
fs.copydir("./source", "./dest")

Metadata & Info

# Get file size in bytes
let bytes = fs.size("video.mp4")

# Get detailed information
let info = fs.info("photo.jpg")
say("Size: " + $info.size)
say("Is File: " + $info.is_file)
say("Is Dir: " + $info.is_dir)
say("Read Only: " + $info.readonly)

Search (Glob)

Find files matching a specific pattern or extension.

# List all .as files in current directory
let scripts = fs.glob("*.as")
for (script in scripts):
    say("Script: " + script)
end
